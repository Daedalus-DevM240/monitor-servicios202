import socket

def verificar_servicio(host, puerto):
    # Creamos un socket TCP
    a_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    # Establecemos un tiempo de espera (timeout) corto (ej. 1 segundo)
    a_socket.settimeout(1)
    
    # Intentamos conectar
    resultado = a_socket.connect_ex((host, int(puerto)))
    
    a_socket.close()
    
    # Si el resultado es 0, el puerto está abierto
    return resultado == 0
